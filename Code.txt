-- Multi-Slider Construction Engine
local sliders = {}
local function createSlider(name, minVal, maxVal, startVal, yPos)
    local Label = Instance.new("TextLabel")
    Label.Size = UDim2.new(0, 160, 0, 15); Label.Position = UDim2.new(0, 10, 0, yPos); Label.BackgroundTransparency = 1
    Label.Text = name .. ": " .. startVal; Label.TextColor3 = Color3.fromRGB(255, 255, 255); Label.Font = Enum.Font.SourceSans
    Label.TextSize = 12; Label.Parent = MainFrame

    local Track = Instance.new("Frame")
    Track.Size = UDim2.new(0, 160, 0, 6); Track.Position = UDim2.new(0, 10, 0, yPos + 23)
    Track.BackgroundColor3 = Color3.fromRGB(40, 40, 40); Track.BorderSizePixel = 0; Track.Parent = MainFrame

    local Handle = Instance.new("TextButton")
    Handle.Size = UDim2.new(0, 12, 0, 16); Handle.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
    Handle.BorderColor3 = Color3.fromRGB(150, 150, 150); Handle.Text = ""; Handle.Parent = Track
    table.insert(handles, Handle)

    local data = {Min = minVal, Max = maxVal, Val = startVal, Track = Track, Handle = Handle, Label = Label, Name = name, Dragging = false}
    Handle.Position = UDim2.new((startVal - minVal) / (maxVal - minVal), 0, 0, -5)
    
    Handle.MouseButton1Down:Connect(function() 
        data.Dragging = true 
    end)
    
    table.insert(sliders, data)
end

createSlider("Walkspeed", 16, 1000, 16, 290)
createSlider("Fly Speed", 20, 1000, 60, 335)
createSlider("Jump Height", 7, 1000, 7, 380)

sliders[2].Handle.Position = UDim2.new((60 - 20) / (1000 - 20), -6, 0, -5)

UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType ~= Enum.UserInputType.MouseMovement and input.UserInputType ~= Enum.UserInputType.Touch then return end
    for _, s in ipairs(sliders) do
        if s.Dragging then
            local pct = math.clamp((input.Position.X - s.Track.AbsolutePosition.X) / s.Track.AbsoluteSize.X, 0, 1)
            s.Handle.Position = UDim2.new(pct, -6, 0, -5)
            s.Val = math.round(s.Min + (pct * (s.Max - s.Min)))
            s.Label.Text = s.Name .. ": " .. s.Val
        end
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        for _, s in ipairs(sliders) do 
            s.Dragging = false 
        end
    end
end)

-- Attribute Continuous Loop & FPS Calculation
local characterLoopConnection
local lastTime = os.clock()
local frameCount = 0

local function maintainStats()
    if characterLoopConnection then characterLoopConnection:Disconnect() end
    characterLoopConnection = RunService.Heartbeat:Connect(function()
        local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        if hum then
            hum.WalkSpeed, hum.JumpHeight = sliders[1].Val, sliders[3].Val
            local est = math.sqrt(2 * workspace.Gravity * sliders[3].Val)
            if tostring(est) ~= "nan" then hum.JumpPower = est end
        end

        frameCount = frameCount + 1
        local currentTime = os.clock()
        if currentTime - lastTime >= 0.5 then
            local fps = math.round(frameCount / (currentTime - lastTime))
            FpsLabel.Text = "FPS: " .. tostring(fps)
            frameCount = 0
            lastTime = currentTime
        end
    end)
end
maintainStats()

-- ESP Module
local espEnabled = false
local function applyESP(p, char)
    if p == LocalPlayer then return end
    local old = char:FindFirstChild("ESPHighlight") if old then old:Destroy() end
    if not espEnabled or not char:WaitForChild("Humanoid", 5) then return end
    
    local hl = Instance.new("Highlight")
    hl.Name = "ESPHighlight"; hl.Adornee = char; hl.FillColor = Color3.fromRGB(65, 0, 125)
    hl.OutlineColor = Color3.fromRGB(255, 255, 255); hl.FillTransparency = 0.4; hl.OutlineTransparency = 0
    hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop; hl.Parent = char
end

local function setupPlayer(p)
    p.CharacterAdded:Connect(function(c) applyESP(p, c) end)
    if p.Character then applyESP(p, p.Character) end
end

for _, p in ipairs(Players:GetPlayers()) do setupPlayer(p) end
Players.PlayerAdded:Connect(setupPlayer)

-- Fly Module
local flying, lv, att, flyConnection = false, nil, nil, nil
local function stopFly()
    flying = false
    FlyBtn.Text, FlyBtn.TextColor3 = "Fly: OFF", Color3.fromRGB(255, 0, 0)
    if flyConnection then flyConnection:Disconnect() flyConnection = nil end
    if lv then lv:Destroy() lv = nil end
    if att then att:Destroy() att = nil end
    local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
    if hum then hum.PlatformStand = false end
end

FlyBtn.MouseButton1Click:Connect(function()
    if flying then stopFly() return end
    local char = LocalPlayer.Character local root = char and char:WaitForChild("HumanoidRootPart", 5) local hum = char and char:FindFirstChildOfClass("Humanoid")
    if not root or not hum then return end
    
    hum.PlatformStand, flying = true, true
    FlyBtn.Text, FlyBtn.TextColor3 = "Fly: ON", Color3.fromRGB(0, 255, 0)
    att = Instance.new("Attachment", root)
    lv = Instance.new("LinearVelocity", root)
    lv.Attachment0, lv.MaxForce, lv.VectorVelocity = att, math.huge, Vector3.zero

    flyConnection = RunService.RenderStepped:Connect(function()
        if not flying or not lv then return end
        local dir = Vector3.zero
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then dir += Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then dir -= Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then dir -= Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then dir += Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then dir += Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then dir -= Vector3.new(0, 1, 0) end
        lv.VectorVelocity = dir.Magnitude > 0 and dir.Unit * sliders[2].Val or Vector3.zero
    end)
end)

LocalPlayer.CharacterAdded:Connect(function()

    if noclipConnection then noclipConnection:Disconnect() noclipConnection = nil end

        end
    end

    if sayMessageRequest and sayMessageRequest:IsA("RemoteEvent") then
        sayMessageRequest:FireServer(message, "All")
    end
end)
